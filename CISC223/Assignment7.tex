	Fact (i): {\fontfamily{qcr}\selectfont x == y*(y+1)/2 implies x+(y+1) == (y+1)*((y+1)+1)/2}
	\\Proof:
		\begin{align*}
			x &= \frac{y(y+1)}{2}\\
			x + (y + 1) &= \frac{y(y+1)}{2} + y + 1\\
			x + (y + 1) &= \frac{y(y+1)+2y+2}{2}\\
			x + (y + 1) &= \frac{y^2 + 3y+2}{2}\\
			x + (y + 1) &= \frac{(y+1)(y+2)}{2}\\
			x + (y + 1) &= \frac{(y+1)((y+1)+1)}{2}	
		\end{align*}

	Fact(ii): {\fontfamily{qcr}\selectfont y == 6 \&\& z < -1 implies y + 2z <= 5}

	\begin{align*}
		y &= 6\\
		z &< -1\\
		z &\leq -2\\

		y + 2z &\leq y - 4\\
		y + 2z &\leq 6 - 4\\
		y + 2z &\leq 2\\
		y + 2z &\leq 5
	\end{align*}
